\section{Data Dictionary}\label{data-dictionary}}

\texttt{age} - Age of the patient

\texttt{sex} - Sex of the patient

\texttt{cp} - Chest pain type \textasciitilde{} 0 = Typical Angina, 1 =
Atypical Angina, 2 = Non-anginal Pain, 3 = Asymptomatic

\texttt{trtbps} - Resting blood pressure (in mm Hg)

\texttt{chol} - Cholestoral in mg/dl fetched via BMI sensor

\texttt{fbs} - (fasting blood sugar \textgreater{} 120 mg/dl)
\textasciitilde{} 1 = True, 0 = False

\texttt{restecg} - Resting electrocardiographic results
\textasciitilde{} 0 = Normal, 1 = ST-T wave normality, 2 = Left
ventricular hypertrophy

\texttt{thalachh} - Maximum heart rate achieved

\texttt{oldpeak} - Previous peak
